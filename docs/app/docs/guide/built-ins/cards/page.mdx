export const metadata = {
  sidebarTitle: 'Cards'
}

import { CardsIcon, OneIcon, WarningIcon } from '@components/icons'
import { useMDXComponents } from 'nextra-theme-docs'
import { compileMdx } from 'nextra/compile'
import { Cards, RemoteContent } from 'nextra/components'

# Cards Component

export async function CardsPage() {
  const mdx = `
<Cards>
  <Cards.Card
    icon={<WarningIcon />}
    title="Callout"
    href="/docs/guide/built-ins/callout"
  />
  <Cards.Card
    icon={<CardsIcon />}
    title="Tabs"
    href="/docs/guide/built-ins/tabs"
  />
  <Cards.Card
    icon={<OneIcon />}
    title="Steps"
    href="/docs/guide/built-ins/steps"
  />
</Cards>`
  const { result } = await compileMdx(`
## Example

A built-in component to embed a link in a card with an icon.

${mdx}

## Usage

First, import the \`Card\` and \`Cards\` components. Then, import the icons that you
want to use. To create a set of cards, follow the example below.

~~~mdx filename="MDX"
import { Cards } from 'nextra/components'
import { CardsIcon, OneIcon, WarningIcon } from '../path/with/your/icons'
${mdx}
~~~`)
  const components = useMDXComponents({ Cards, CardsIcon, OneIcon, WarningIcon })
  return <RemoteContent compiledSource={result} components={components}  />
}

<CardsPage />
